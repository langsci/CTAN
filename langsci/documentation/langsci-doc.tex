\option{output}{different output formats}{short}
\begin{itemize}
\item book: The book with frontcover and backcover 
\item inprep: Mark manuscript as in preparation
\item paper: contribution to edited volume
\item guidelines: Choose color grey; condense frontmatter
\item coverbod: create a two-page wide cover  (back-spine-front) for use with BoD
\item covercreatespace: create a two-page wide cover  (back-spine-front) for use with CreateSpace 
\end{itemize}
